    <script>
        function fillSearch(val) {
            document.getElementById('searchInput').value = val;
            performSearch();
        }

        async function performSearch() {
            const query = document.getElementById('searchInput').value;
            const resultsDiv = document.getElementById('results');
            if (!query) return;

            resultsDiv.innerText = "جاري البحث في أمازون ومواقع القطع... ⏳";

            try {
                const response = await fetch('/api/search', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ query })
                });
                const data = await response.json();
                resultsDiv.innerText = data.result || "لم يتم العثور على نتائج.";
            } catch (err) {
                resultsDiv.innerText = "خطأ: تأكد أن السيرفر شغال.";
            }
        }

        document.getElementById('imageInput').addEventListener('change', async (e) => {
            const file = e.target.files[0];
            const status = document.getElementById('status');
            if (!file) return;

            status.innerText = "جاري قراءة البيانات من الصورة... 🔍";
            
            Tesseract.recognize(file, 'eng', { logger: m => console.log(m) })
                .then(({ data: { text } }) => {
                    if(text.trim().length > 2) {
                        status.innerText = "تم استخراج النص! جاري البحث...";
                        document.getElementById('searchInput').value = text.trim();
                        performSearch();
                    } else {
                        status.innerText = "لم أستطع قراءة نص واضح، جرب صورة أوضح.";
                    }
                })
                .catch(err => {
                    status.innerText = "فشل في معالجة الصورة.";
                });
        });
    </script>
